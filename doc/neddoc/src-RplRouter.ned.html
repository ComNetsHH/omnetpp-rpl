package rpl;

import inet.node.inet.AdhocHost;
import rpl.Rpl;
import rpl.TrickleTimer; 

module RplRouter extends AdhocHost
{
    submodules:
        rpl: Rpl {
            @display("p=825,226");
        }
        trickleTimer: TrickleTimer {
        }
        
    connections:
        rpl.ipOut --> tn.in++;
        rpl.ipIn <-- tn.out++;
        rpl.ttModule <--> trickleTimer.rpModule; 
}
