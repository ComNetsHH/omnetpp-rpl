45 Dio* ObjectiveFunction::getPreferredParent(std::map<Ipv6Address, Dio *> candidateParents) {
46  // determine parent with lowest rank
47  if (candidateParents.empty()) {
48  EV_WARN << "Couldn't determine preferred parent, provided set is empty" << endl;
49  return nullptr;
50  }
51 
52  for (auto cp : candidateParents)
53  EV_DETAIL << cp.first << " - " << cp.second << endl;
54 
55  Dio *preferredParent = candidateParents.begin()->second;
56  uint16_t minRank = preferredParent->getRank();
57  for (std::pair<Ipv6Address, Dio *> candidate : candidateParents) {
58  uint16_t candidateParentRank = candidate.second->getRank();
59  if (candidateParentRank < minRank) {
60  minRank = candidateParentRank;
61  preferredParent = candidate.second;
62  }
63  }
64 
65  return preferredParent;
66 }

68 uint16_t ObjectiveFunction::calcRank(Dio* preferredParent) {
69  if (!preferredParent)
70  throw cRuntimeError("Cannot calculate rank, preferredParent argument is null");
71 
72  uint16_t prefParentRank = preferredParent->getRank();
74  switch (type) {
75  case HOP_COUNT:
76  return prefParentRank + 1;
77  default:
78  return prefParentRank + minHopRankIncrease;
79  }
