254 void Rpl::processPacket(Packet *packet)
255 {
256  if (floating) {
257  EV_DETAIL << "Not processing packet " << packet
258  << "\n due to floating (detached) state" << endl;
259  delete packet;
260  return;
261  }
262  auto rplHeader = packet->popAtFront<RplHeader>();
263  auto rplBody = packet->peekData<RplPacket>();
264  auto senderAddr = rplBody->getSrcAddress();
265  switch (rplHeader->getIcmpv6Code()) {
266  case DIO: {
267  processDio(dynamicPtrCast<const Dio>(rplBody));
268  break;
269  }
270  case DAO: {
271  if (daoEnabled)
272  processDao(dynamicPtrCast<const Dao>(rplBody));
273  else
274  EV_DETAIL << "DAO support not enabled, discarding packet" << endl;
275  break;
276  }
277  case DAO_ACK: {
278  processDaoAck(dynamicPtrCast<const Dao>(rplBody));
279  break;
280  }
281  case DIS: {
282  processDis(dynamicPtrCast<const Dis>(rplBody));
283  break;
284  }
285  default: EV_WARN << "Unknown Rpl packet" << endl;
286  }
287 
288  delete packet;
289 }

303 void Rpl::sendRplPacket(const Ptr<RplPacket>& body, RplPacketCode code, const L3Address& nextHop, double delay)
304 {
305  Packet *pkt = new Packet("inet::RplPacket");
306  auto header = makeShared<RplHeader>();
307  header->setIcmpv6Code(code);
308  pkt->addTag<PacketProtocolTag>()->setProtocol(&Protocol::manet);
309  pkt->addTag<DispatchProtocolReq>()->setProtocol(&Protocol::ipv6);
310  if (interfaceEntryPtr)
311  pkt->addTag<InterfaceReq>()->setInterfaceId(interfaceEntryPtr->getInterfaceId());
312  auto addresses = pkt->addTag<L3AddressReq>();
313  addresses->setSrcAddress(getSelfAddress());
314  addresses->setDestAddress(nextHop);
315  pkt->insertAtFront(header);
316  pkt->insertAtBack(body);
317  sendPacket(pkt, delay);
318 }

350 void Rpl::processDio(const Ptr<const Dio>& dio)
351 {
352  EV_DETAIL << "Processing DIO from " << dio->getSrcAddress() << endl;
353  emit(dioReceivedSignal, dio->dup());
354  // If node's not a part of any DODAG, join the one advertised
355  if (!isRoot && dodagId == Ipv6Address::UNSPECIFIED_ADDRESS && dio->getRank() != INF_RANK) {
356  dodagId = dio->getDodagId();
357  dodagVersion = dio->getDodagVersion();
358  instanceId = dio->getInstanceId();
359  storing = dio->getStoring();
360  dtsn = dio->getDtsn();
361  EV_DETAIL << "Joined new DODAG with id - " << dodagId << endl;
362  // Start broadcasting DIOs, diffusing DODAG control data
363  if (trickleTimer->hasStarted())
364  trickleTimer->reset();
365  else
366  trickleTimer->start();
367  } else {
373  if (checkPoisonedParent(dio)) {
374  EV_DETAIL << "Received poisoned DIO from preferred parent - "
375  << preferredParent->getSrcAddress() << endl;
376  deletePrefParent(true);
377  updatePreferredParent();
378  return;
379  }
380  }
381  trickleTimer->ctrlMsgReceived();
382  // Do not process DIO from unknown DAG/RPL instance or if receiver is root
383  if (checkUnknownDio(dio) || isRoot) {
384  EV_DETAIL << "Unknown DODAG/InstanceId, or receiver is root - discarding DIO" << endl;
385  return;
386  }
392  if (dio->getDodagVersion() > dodagVersion) {
393  EV_DETAIL << "Newer DODAG version discovered, joining and clearing former parent sets" << endl;
394  backupParents.erase(backupParents.begin(), backupParents.end());
395  candidateParents.erase(candidateParents.begin(), candidateParents.end());
396  dodagVersion = dio->getDodagVersion();
397  } else if (dio->getDodagVersion() < dodagVersion)
398  EV_DETAIL << "Outdated DODAG version advertised, discarding DIO" << endl;
403  if (dio->getDtsn() > dtsn) {
404  EV_DETAIL << "DTSN incremented, unicast DAO scheduled" << endl;
405  dtsn = dio->getDtsn();
406  sendRplPacket(createDao(), DAO, preferredParent->getSrcAddress(), daoDelay);
407  }
408  if (dio->getRank() > rank) {
409  EV_DETAIL << "Higher rank advertised, discarding DIO" << endl;
410  return;
411  }
412  addNeighbour(dio);
413  updatePreferredParent();
414 }

421 void Rpl::processDao(const Ptr<const Dao>& dao) {
422  emit(daoReceivedSignal, dao->dup());
423  auto daoSender = dao->getSrcAddress();
424  auto advertisedDest = dao->getReachableDest();
425  if (dao->getDaoAckRequired()) {
426  sendRplPacket(createDao(Ipv6Address::UNSPECIFIED_ADDRESS), DAO_ACK, daoSender);
427  EV_DETAIL << "DAO_ACK sent back to " << daoSender << endl;
428  }
429 
434  if (storing || isRoot) {
435  if (!checkDestKnown(daoSender, advertisedDest)) {
436  updateRoutingTable(dao->dup());
437  EV_DETAIL << "Destination learned from DAO - " << advertisedDest
438  << " reachable via " << daoSender << endl;
439  }
440  }
444  if (!isRoot && preferredParent && !matchesSuffix(daoSender, preferredParent->getSrcAddress())) {
445  sendRplPacket(createDao(advertisedDest), DAO,
446  preferredParent->getSrcAddress(), daoDelay * uniform(1, 2));
447  EV_DETAIL << "Forwarding DAO upwards to " << preferredParent->getSrcAddress()
448  << " advertising " << advertisedDest << " reachability" << endl;
449  }
450 }

464 void Rpl::updatePreferredParent()
465 {
466  Dio *newPrefParent;
467  EV_DETAIL << "Choosing preferred parent from "
468  << boolStr(candidateParents.empty() && par("useBackupAsPreferred").boolValue(),
469  "backup", "candidate") << " parent set" << endl;
474  if (candidateParents.empty()) {
475  if (par("useBackupAsPreferred").boolValue())
476  newPrefParent = objectiveFunction->getPreferredParent(backupParents);
477  else {
478  detachFromDodag();
479  return;
480  }
481 
482  } else
483  newPrefParent = objectiveFunction->getPreferredParent(candidateParents);
484 
485  auto newPrefParentAddr = newPrefParent->getSrcAddress();
490  if (checkPrefParentChanged(newPrefParentAddr)) {
491  emit(parentChangedSignal, newPrefParent->dup());
492  updateRoutingTable(newPrefParentAddr);
493  EV_DETAIL << "Updated preferred parent to - " << newPrefParentAddr << endl;
498  trickleTimer->reset();
499  if (daoEnabled) {
500  sendRplPacket(createDao(), DAO, newPrefParentAddr,
501  daoDelay * uniform(1, 2));
502  }
503 
504  }
505  preferredParent = newPrefParent->dup();
507  rank = objectiveFunction->calcRank(preferredParent);
508  EV_DETAIL << "Rank - " << rank << endl;
509 }

573  EV_DETAIL << "Determining packet forwarding direction:\n destination - " << dest << endl;

599 INetfilter::IHook::Result Rpl::checkRplPacketInfo(Packet *datagram) {
600  const auto& networkHeader = findNetworkProtocolHeader(datagram);
601  EV_DETAIL << "Packet source address - " << networkHeader->getSourceAddress().toIpv6() << endl;
602  Ipv6Address *dest = new Ipv6Address();
603  if (networkHeader != nullptr)
604  *dest = networkHeader->getDestinationAddress().toIpv6();
605  auto datagramName = std::string(datagram->getFullName());
606  if (datagramName.find(std::string("Udp")) != std::string::npos) {
607  if (preferredParent == nullptr
608  || dodagId.matches(Ipv6Address::UNSPECIFIED_ADDRESS, prefixLength))
609  {
610  EV_DETAIL << "Not belonging to a dodag, skip RPI header check" << endl;
611  return ACCEPT;
612  }
613  EV_DETAIL << "Checking RPI header for packet " << datagram << endl;
614  RplPacketInfo *rpi;
615  try {
616  rpi = const_cast<RplPacketInfo *> (datagram->popAtBack<RplPacketInfo>(B(4)).get());
617  }
618  catch (std::exception &e) {
619  EV_WARN << "No RPL Packet Information header found in UDP datagram, appending" << endl;
620  try {
621  appendRplPacketInfo(datagram);
622  return ACCEPT;
623  }
624  catch (...) {
625  return DROP;
626  }
627  }
628  // check for rank inconsistencies
629  EV_DETAIL << "Rank error - " << rpi->getRankError() << endl;
630  bool rankInconsistency = checkRankError(rpi);
631  if (rpi->getRankError() && rankInconsistency) {
632  EV_WARN << "Repeated rank error detected for packet "
633  << datagram << "\n dropping..." << endl;
634  return DROP;
635  }
636  if (rankInconsistency)
637  rpi->setRankError(rankInconsistency);
638  EV_DETAIL << "Rank error after check - " << rpi->getRankError() << endl;
644  rpi->setFwdError(!isRoot && checkForwardingError(rpi, dest));
645  if (rpi->getFwdError()) {
646  EV_WARN << "Forwarding error detected for packet " << datagram
647  << "\n destined to " << dest << ", dropping" << endl;
648  return DROP;
649  }
650  // update packet forwarding direction if storing mode is in use
651  // e.g. if unicast P2P packet reaches sub-dodag root with 'O' flag cleared,
652  // and this root can route packet downwards to destination, 'O' flag has to be set.
653  if (storing)
654  rpi->setDown(isRoot || packetTravelsDown(datagram));
655  // try make new shared ptr chunk RPI, to be refactored into reusing existing RPI object
656  auto rpiCopy = makeShared<RplPacketInfo>();
657  rpiCopy->setChunkLength(B(4));
