53 void TrickleTimer::start(bool warmupDelay) {
54  Enter_Method("TrickleTimer::start()");
55  EV_INFO << "Trickle timer started" << endl;
56  started = true;
57  minInterval = DEFAULT_DIO_INTERVAL_MIN;
58  currentInterval = warmupDelay ? minInterval * 4 : minInterval;
59  maxInterval = minInterval * (pow(2, numDoublings));
60  ctrlMsgReceivedCtn = 0;
61 
62  intervalTriggerEvent = new cMessage("Trickle timer current interval ended",
63  TRICKLE_INTERVAL_UPDATE_EVENT);
64  trickleTriggerEvent = new cMessage("Trickle timer trigger self-msg",
65  TRICKLE_TRIGGER_EVENT);
66 
67  scheduleAt(simTime() + currentInterval, intervalTriggerEvent);
68  scheduleNext();
69 }

85 void TrickleTimer::processSelfMessage(cMessage *message)
86 {
87  switch (message->getKind()) {
88  case TRICKLE_INTERVAL_UPDATE_EVENT: {
89  if (currentInterval < maxInterval)
90  currentInterval *= 2;
91  ctrlMsgReceivedCtn = 0;
92  scheduleAt(simTime() + currentInterval, intervalTriggerEvent);
93  scheduleNext();
94  EV_INFO << "Trickle interval doubled, current = " <<
95  currentInterval << endl;
96  break;
97  }
98  case TRICKLE_TRIGGER_EVENT: {
99  send(new cMessage("", TRICKLE_TRIGGER_EVENT), "rpModule$o");
100  break;
101  }
102  default: {
103  throw cRuntimeError("Unknown kind of trickle timer self message");
104  }
105  }
106 }

130 void TrickleTimer::reset() {
131  Enter_Method_Silent("TrickleTimer::reset()");
132  ctrlMsgReceivedCtn = 0;
133  currentInterval = minInterval;
134  try {
135  cancelEvent(intervalTriggerEvent);
136  cancelEvent(trickleTriggerEvent);
137  scheduleAt(simTime() + currentInterval, intervalTriggerEvent);
138  EV_DETAIL << "Trickle timer reset" << endl;
139  }
140  catch (std::exception &e) {
141  EV_WARN << "Exception: " << e.what() <<" while resetting trickle timer, currentInterval = "
142  << currentInterval << "; \n next interval doubling would be scheduled at "
143  << simTime() + currentInterval << endl;
144  }
145  scheduleNext();
146 }
